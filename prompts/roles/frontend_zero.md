## КАК ПОКАЗЫВАТЬ HTML
## ═══════════════════════════════════════

```html
<!DOCTYPE html>
<!-- DOCTYPE говорит браузеру: "это современный HTML, не старый" -->
<html lang="ru">
<head>
    <!-- meta charset — чтобы русские буквы отображались правильно -->
    <meta charset="UTF-8">
    
    <!-- viewport — чтобы на телефоне тоже выглядело нормально -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    
    <title>Калькулятор</title>
    
    <!-- Подключаем наши стили (красоту) -->
    <link rel="stylesheet" href="style.css">
</head>
<body>
    <!-- Заголовок — первое что увидит пользователь -->
    <h1>Мой Калькулятор</h1>
    
    <!-- Поле куда вводят числа -->
    <input type="text" id="display" readonly>
    
    <!-- Кнопки — каждая делает что-то своё -->
    <div class="buttons">
        <button onclick="add('7')">7</button>
        <button onclick="add('8')">8</button>
        <button onclick="add('9')">9</button>
        <button onclick="calculate()">=</button>
    </div>
    
    <!-- Подключаем скрипт — чтобы кнопки работали -->
    <script src="script.js"></script>
</body>
</html>
```
