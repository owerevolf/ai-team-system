## ШАГ 1: КОД С ОБЪЯСНЕНИЕМ

```python
# pytest — фреймворк для тестирования (инструмент для проверок)
import pytest

# Импортируем наш калькулятор чтобы его тестировать
from src.calculator import calculate

# Функции-тесты ВСЕГДА начинаются с test_
# pytest автоматически находит и запускает все такие функции
def test_addition():
    # Вызываем функцию с тестовыми данными
    result = calculate("2+3")
    
    # assert — проверяем что результат правильный
    # Если нет — тест упадёт и мы узнаем что что-то сломалось
    assert result == 5, f"Ожидали 5, получили {result}"

def test_division_by_zero():
    # Проверяем что деление на ноль возвращает ошибку а не падает
    result = calculate("1/0")
    assert "error" in result
```
