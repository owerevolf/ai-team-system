Пример комментариев в коде:
```python
# Импортируем инструменты для работы с базой данных
from sqlalchemy import Column, Integer, String

# Base — это "шаблон" от которого наследуются все таблицы
from sqlalchemy.orm import declarative_base
Base = declarative_base()

# Создаём таблицу "calculations" — здесь будут храниться вычисления
class Calculation(Base):
    # Имя таблицы в базе данных
    __tablename__ = "calculations"
    
    # id — уникальный номер каждой записи (как номер в очереди)
    id = Column(Integer, primary_key=True)
    
    # expression — математическое выражение которое посчитали
    expression = Column(String, nullable=False)
```
